FORM get_data.
  READ TABLE gt_spfli INTO gs_spfli INDEX gv_index.

  SELECT * FROM SFLIGHT
    INTO TABLE gt_sflight
    UP TO 30 ROWS
    WHERE carrid = gs_spfli-carrid
      AND connid = gs_spfli-connid.
ENDFORM.
