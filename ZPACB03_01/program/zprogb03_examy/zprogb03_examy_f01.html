FORM set_data .
  DATA: gv_name TYPE scustom-name,
        gv_carrname TYPE scarr-carrname,
        gv_status TYPE ZTREQFOOD_B03-status,
        gv_reqdate TYPE ZTREQFOOD_B03-reqdate.
  DATA: gt_table TYPE TABLE OF ZTREQFOOD_B03,
        gs_table TYPE ZTREQFOOD_B03.
  DATA: gv_index_data TYPE I VALUE 1.

  SELECT * FROM sbook
    INTO CORRESPONDING FIELDS OF TABLE gt_data
    UP TO 100 ROWS
    WHERE carrid = pa_car
      AND connid IN pa_con
      AND fldate IN pa_fld
      AND cancelled NE 'X'.
  IF sy-subrc <> 0.
    MESSAGE '## ### #### ####.' TYPE 'E'.
  ELSE.
    LOOP AT gt_data INTO gs_data.
      SELECT SINGLE name FROM scustom
        INTO gv_name
        WHERE id = gs_data-customid.
      SELECT SINGLE carrname FROM scarr
        INTO gv_carrname
        WHERE carrid = gs_data-carrid.
      SELECT SINGLE status FROM ZTREQFOOD_B03
        INTO gv_status
        WHERE carrid = gs_data-carrid
          AND connid = gs_data-connid
          AND fldate = gs_data-fldate
          AND bookid = gs_data-bookid.
      IF sy-subrc <> 0.
        gs_data-status = 'A'.
        gs_data-name = gv_name.
        gs_data-carrname = gv_carrname.
        MODIFY gt_data FROM gs_data TRANSPORTING name carrname status.
      ELSE.
        SELECT * FROM ZTREQFOOD_B03
          INTO CORRESPONDING FIELDS OF TABLE gt_table
          WHERE carrid = gs_data-carrid
             AND connid = gs_data-connid
             AND fldate = gs_data-fldate
             AND bookid = gs_data-bookid
          ORDER BY reqdate DESCENDING.
        READ TABLE gt_table INTO gs_table INDEX 1.
        gs_data-status = gs_table-status.
        gs_data-reqdate = gs_table-reqdate.
        gs_data-name = gv_name.
        gs_data-carrname = gv_carrname.
        MODIFY gt_data FROM gs_data TRANSPORTING name carrname status reqdate.
      ENDIF.
      CLEAR: gv_name, gv_carrname, gs_data, gv_status, gv_reqdate.
    ENDLOOP.
    SORT gt_data BY bookid.

*    SELECT * FROM ztreqfood_b03
*      INTO CORRESPONDING FIELDS OF TABLE gt_table.
*
*    PERFORM set_data_plus.
  ENDIF.
ENDFORM.

FORM set_fixedvalue .
  DATA: lt_domain_values TYPE TABLE OF DD07V,
        ls_domain_value LIKE LINE OF lt_domain_values.
  CALL FUNCTION 'GET_DOMAIN_VALUES'
    EXPORTING
      domname               = 'ZDOREQSTATUS'
    TABLES
      VALUES_TAB            = lt_domain_values
    EXCEPTIONS
      NO_VALUES_FOUND       = 1
      OTHERS                = 2.
  IF sy-subrc <> 0.
    MESSAGE '#### fixed value #### ##' TYPE 'E'.
  ELSE.
    DATA: gv_statpos TYPE NUMC4.
    CLEAR: gs_data.
    LOOP AT gt_data INTO gs_data.
      IF gs_data-status = 'A'.
        gv_statpos = 1.
      ELSEIF gs_data-status = 'B'.
        gv_statpos = 2.
      ELSEIF gs_data-status = 'C'.
        gv_statpos = 3.
      ENDIF.
      READ TABLE lt_domain_values INTO ls_domain_value
        WITH KEY valpos = gv_statpos.
      CHECK sy-subrc = 0.
      gs_data-status_txt = ls_domain_value-ddtext.
      MODIFY gt_data FROM gs_data TRANSPORTING status_txt.
    ENDLOOP.
  ENDIF.

ENDFORM.

FORM set_data2 .
  DATA: gv_apptxt TYPE S_MEALTEXT,
        gv_maintxt TYPE S_MEALTEXT,
        gv_desstxt TYPE S_MEALTEXT.
  CLEAR gs_data.

  READ TABLE gt_data INTO gs_data INDEX gv_index.
  MOVE-CORRESPONDING gs_data TO SBOOK.

  SELECT * FROM smenu
    INTO CORRESPONDING FIELDS OF TABLE gt_smenu
    WHERE carrid = gs_data-carrid.

  LOOP AT gt_smenu INTO gs_smenu.
    gs_smenu-carrname = gs_data-carrname.
    SELECT SINGLE TEXT FROM smealt
      INTO gv_apptxt
      WHERE carrid = gs_smenu-carrid
        AND MEALNUMBER = gs_smenu-STARTER.
    SELECT SINGLE TEXT FROM smealt
      INTO gv_maintxt
      WHERE carrid = gs_smenu-carrid
        AND MEALNUMBER = gs_smenu-MAINCOURSE.
    SELECT SINGLE TEXT FROM smealt
      INTO gv_desstxt
      WHERE carrid = gs_smenu-carrid
        AND MEALNUMBER = gs_smenu-DESSERT.
    gs_smenu-app_txt = gv_apptxt.
    gs_smenu-main_txt = gv_maintxt.
    gs_smenu-dess_txt = gv_desstxt.
    MODIFY gt_smenu FROM gs_smenu TRANSPORTING carrname app_txt main_txt dess_txt.
  ENDLOOP.

  SORT gt_smenu BY menunumber.
ENDFORM.

FORM update_reqtable .
  CLEAR: gs_data, gs_smenu, gs_req.
  DATA: gv_reqdate TYPE ZTREQFOOD_B03-reqdate.
  DATA: gt_temp TYPE TABLE OF ZTREQFOOD_B03.

  READ TABLE gt_data INTO gs_data INDEX gv_index.
*  gv_reqdate = ZTREQFOOD_B03-REQDATE.

  SELECT * FROM ZTREQFOOD_B03
    INTO TABLE gt_temp
    WHERE carrid = gs_data-carrid
      AND connid = gs_data-connid
      AND fldate = gs_data-fldate
      AND bookid = gs_data-bookid
    ORDER BY reqdate DESCENDING.

  IF sy-subrc = 0.
    READ TABLE gt_temp INTO gs_req INDEX 1.
    IF gs_req-status = 'B' AND gs_req-reqdate NE sy-datum.
      gs_req-status = 'C'.
      gs_req-enddate = sy-datum - 1.
      UPDATE ZTREQFOOD_B03 FROM gs_req.
      PERFORM set_data.

      CLEAR: gs_req, gs_smenu.
      READ TABLE gt_smenu INTO gs_smenu INDEX gv_index_req.
      gs_req-carrid = gs_data-carrid.
      gs_req-connid = gs_data-connid.
      gs_req-fldate = gs_data-fldate.
      gs_req-bookid = gs_data-bookid.
      gs_req-status = 'B'.
      gs_req-REQDATE = sy-datum.
      gs_req-MENUNUMBER = gs_smenu-MENUNUMBER.
      gs_req-ENDDATE = '99991231'.
      INSERT ZTREQFOOD_B03 FROM gs_req.
      gs_data-reqdate = sy-datum.
      gs_data-status = 'B'.
      APPEND gs_data TO gt_data.
      PERFORM set_fixedvalue.
    ELSEIF gs_req-status = 'C' AND gs_req-reqdate EQ sy-datum.
      CLEAR: gs_req, gs_smenu.
      READ TABLE gt_smenu INTO gs_smenu INDEX gv_index_req.
      gs_req-carrid = gs_data-carrid.
      gs_req-connid = gs_data-connid.
      gs_req-fldate = gs_data-fldate.
      gs_req-bookid = gs_data-bookid.
      gs_req-status = 'B'.
      gs_req-REQDATE = sy-datum.
      gs_req-MENUNUMBER = gs_smenu-MENUNUMBER.
      gs_req-ENDDATE = '99991231'.
      UPDATE ZTREQFOOD_B03 FROM gs_req.
      PERFORM set_data.
      PERFORM set_fixedvalue.
    ENDIF.
  ELSE.
    CLEAR: gs_req, gs_smenu.
    READ TABLE gt_smenu INTO gs_smenu INDEX gv_index_req.
    gs_req-carrid = gs_data-carrid.
    gs_req-connid = gs_data-connid.
    gs_req-fldate = gs_data-fldate.
    gs_req-bookid = gs_data-bookid.
    gs_req-status = 'B'.
    gs_req-REQDATE = sy-datum.
    gs_req-MENUNUMBER = gs_smenu-MENUNUMBER.
    gs_req-ENDDATE = '99991231'.
    INSERT ZTREQFOOD_B03 FROM gs_req.
    PERFORM set_data.
    PERFORM set_fixedvalue.
  ENDIF.
ENDFORM.

FORM set_data_plus .
  DATA: ls_data TYPE ZTREQFOOD_B03,
        lt_data TYPE TABLE OF ZTREQFOOD_B03.
  DATA: lv_index TYPE I.
  SELECT * FROM  ZTREQFOOD_B03
    INTO CORRESPONDING FIELDS OF TABLE lt_data.
  LOOP AT gt_data INTO gs_data.
    LOOP AT lt_data INTO ls_data.
      IF gs_data-carrid = ls_data-carrid AND gs_data-connid = ls_data-connid
        AND gs_data-fldate = ls_data-fldate AND gs_data-bookid = ls_data-bookid
        AND gs_data-reqdate NE ls_data-reqdate.
        gs_data-reqdate = ls_data-reqdate.
        gs_data-status = ls_data-status.
        APPEND gs_data TO gt_data.
      ENDIF.
    ENDLOOP.
  ENDLOOP.
ENDFORM.
