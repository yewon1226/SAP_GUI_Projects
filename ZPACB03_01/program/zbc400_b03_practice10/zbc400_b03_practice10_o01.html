MODULE init_alv OUTPUT.
  IF go_cont IS INITIAL.
    CREATE OBJECT go_cont
      EXPORTING
        container_name              = 'AREA'
      EXCEPTIONS
        cntl_error                  = 1
        cntl_system_error           = 2
        create_error                = 3
        lifetime_error              = 4
        lifetime_dynpro_dynpro_link = 5
        others                      = 6.

    CREATE OBJECT go_alv
      EXPORTING
        i_parent          = go_cont
      EXCEPTIONS
        error_cntl_create = 1
        error_cntl_init   = 2
        error_cntl_link   = 3
        error_dp_create   = 4
        others            = 5.

    gs_layout-sel_mode = 'A'.
    gs_layout-zebra = 'X'.
    PERFORM set_fcat.
    SET HANDLER lcl_event_handler=>on_double_click FOR go_alv.

    CALL METHOD go_alv->set_table_for_first_display
      EXPORTING
        i_structure_name              = 'ZTVENDER_B03'
        is_layout                     = gs_layout
      CHANGING
        it_outtab                     = gt_data
        it_fieldcatalog               = gt_fcat
      EXCEPTIONS
        invalid_parameter_combination = 1
        program_error                 = 2
        too_many_lines                = 3
        others                        = 4.
  ELSE.
    CALL METHOD go_alv->refresh_table_display.
  ENDIF.
ENDMODULE.

MODULE init_alv_200 OUTPUT.
  IF gv_flag = 1.
   LOOP AT SCREEN.
     IF screen-group1 = 'GRP'.
       screen-invisible = 1.
       screen-active    = 0.
       MODIFY SCREEN.
     ENDIF.
   ENDLOOP.
  ELSE.
   LOOP AT SCREEN.
     IF screen-group1 = 'GRP'.
       screen-invisible = 0.
       screen-active    = 1.
       MESSAGE '### ## ## #' TYPE 'S'.
       MODIFY SCREEN.
     ENDIF.
   ENDLOOP.
  ENDIF.
  IF go_cont_200 IS INITIAL.
    CREATE OBJECT go_cont_200
      EXPORTING
        container_name              = 'AREA2'
      EXCEPTIONS
        cntl_error                  = 1
        cntl_system_error           = 2

        lifetime_error              = 4
        lifetime_dynpro_dynpro_link = 5

        error_cntl_create = 1

        error_cntl_link   = 3
        error_dp_create   = 4
