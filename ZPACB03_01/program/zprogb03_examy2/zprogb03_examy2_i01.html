MODULE user_command_0100 INPUT.
  DATA: lt_select_data TYPE LVC_T_ROW,
        ls_select_data TYPE LVC_S_ROW.
  DATA: gs_req_record TYPE ZTREQFOOD2_B03,
        gv_flag_cancel TYPE I VALUE 0.
  CLEAR gs_data.
  CASE OK_CODE.
    WHEN 'CANC'.
      CALL METHOD go_alv->get_selected_rows
        IMPORTING
          et_index_rows = lt_select_data.
      IF sy-subrc = 0.
        LOOP AT lt_select_data INTO ls_select_data.
          READ TABLE gt_data INTO gs_data INDEX ls_select_data-index.
          IF gs_data-status NE 'B' OR gs_data-reqdate NE sy-datum.
            MESSAGE '### ## ### #### #####.' TYPE 'I'.
            gv_flag_cancel = 1.
          ENDIF.
        ENDLOOP.
        IF gv_flag_cancel = 0.
          LOOP AT lt_select_data INTO ls_select_data.
            READ TABLE gt_data INTO gs_data INDEX ls_select_data-index.
            MOVE-CORRESPONDING gs_data TO gs_req_record.
            gs_req_record-status = 'C'.
            gs_req_record-enddate = sy-datum - 1.
*            gs_req_record-menunumber =
*            UPDATE ZTREQFOOD2_B03 from gs_req_record.
          ENDLOOP.
          PERFORM set_data.
          PERFORM set_fixedvalue.
        ENDIF.
      ELSE.
        MESSAGE '## ### ### ## ## #####' TYPE 'I'.
      ENDIF.
    WHEN 'REQUEST'.
      CALL METHOD go_alv->get_selected_rows
        IMPORTING
          et_index_rows = lt_select_data.
      READ TABLE lt_select_data INTO ls_select_data INDEX 1.
      IF sy-subrc = 0.
        READ TABLE lt_select_data INTO ls_select_data INDEX 2.
        IF sy-subrc = 0.
          MESSAGE '#### ### ### #####.' TYPE 'I'.
        ELSE.
          CLEAR ls_select_data.
          READ TABLE lt_select_data INTO ls_select_data INDEX 1.
          gv_index = ls_select_data-index.
          READ TABLE gt_data INTO gs_data INDEX gv_index.
          IF gs_data-status = 'B'.
            MESSAGE '## ### ### ### #####.' TYPE 'I'.
          ENDIF.
          PERFORM set_data2.
          CALL SCREEN 200.
        ENDIF.
      ELSE.
        MESSAGE '## ### ### ### #####' TYPE 'I'.
      ENDIF.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN OTHERS.
  ENDCASE.
ENDMODULE.
