CLASS lcl_event_handler DEFINITION.
  PUBLIC SECTION.
    CLASS-METHODS: ON_TOOLBAR FOR EVENT TOOLBAR
      OF CL_GUI_ALV_GRID IMPORTING E_OBJECT,
        ON_USER_COMMAND FOR EVENT USER_COMMAND
      OF CL_GUI_ALV_GRID IMPORTING E_UCOMM,
        ON_DOUBLE_CLICK FOR EVENT DOUBLE_CLICK
      OF CL_GUI_ALV_GRID IMPORTING ES_ROW_NO.
ENDCLASS.

CLASS lcl_event_handler IMPLEMENTATION.
  METHOD on_toolbar.
    DATA ls_button TYPE STB_BUTTON.
    ls_button-butn_type = 3.
    INSERT ls_button INTO TABLE e_object->mt_toolbar.

    CLEAR ls_button.
    ls_button-function = 'REQUEST_TB'.
    ls_button-butn_type = 0.
    ls_button-text = 'REQUEST'.
    INSERT ls_button INTO TABLE e_object->mt_toolbar.
  ENDMETHOD.

  METHOD ON_USER_COMMAND.
    DATA: lt_select_data TYPE LVC_T_ROW,
          ls_select_data TYPE LVC_S_ROW.
    CLEAR gv_index_req.

    CASE E_UCOMM.
      WHEN 'REQUEST_TB'.
        CALL METHOD go_alv2->get_selected_rows
          IMPORTING
            et_index_rows = lt_select_data.
        READ TABLE lt_select_data INTO ls_select_data INDEX 1.
        IF sy-subrc = 0.
          READ TABLE lt_select_data INTO ls_select_data INDEX 2.
          IF sy-subrc = 0.
            MESSAGE '### ### # ## #####.' TYPE 'I'.
          ELSE.
            gv_index_req = ls_select_data-index.
            PERFORM update_reqtable.
          ENDIF.
        ELSE.
          MESSAGE '## ### #### #####' TYPE 'I'.
        ENDIF.
      WHEN OTHERS.
    ENDCASE.
  ENDMETHOD.

  METHOD ON_DOUBLE_CLICK.
    CLEAR gs_data.
    gv_index_food = es_row_no-row_id.
    READ TABLE gt_data INTO gs_data INDEX gv_index_food.
    IF gs_data-status NE 'B'.
      MESSAGE '## ##### #### ####.' TYPE 'I'.
    ELSE.
      DATA: gvs_menu TYPE smenu,
            gvt_req TYPE TABLE OF ZTREQFOOD2_B03,
            gvs_req TYPE ZTREQFOOD2_B03,
            gv_text_app TYPE CHAR40,
            gv_text_main TYPE CHAR40,
            gv_text_dess TYPE CHAR40.
      SELECT * FROM ztreqfood2_b03
        INTO CORRESPONDING FIELDS OF TABLE gvt_req
        WHERE carrid = gs_data-carrid
          AND connid = gs_data-connid
          AND fldate = gs_data-fldate
          AND bookid = gs_data-bookid
        ORDER BY REQDATE DESCENDING.
      SELECT SINGLE * FROM ztreqfood2_b03
        INTO  gvs_req
        WHERE carrid = gs_data-carrid
          AND connid = gs_data-connid
          AND fldate = gs_data-fldate
          AND bookid = gs_data-bookid.
      SELECT SINGLE * FROM smenu
        INTO gvs_menu
        WHERE carrid = gs_data-carrid
          AND menunumber = gvs_req-menunumber.

      SMENU-STARTER = gvs_menu-starter.
      SMENU-MAINCOURSE = gvs_menu-maincourse.
      SMENU-DESSERT = gvs_menu-dessert.

      SELECT SINGLE text FROM SMEALT
        INTO gv_text_app
        WHERE MEALNUMBER = gvs_menu-starter.
      SELECT SINGLE text FROM SMEALT
        INTO gv_text_main
        WHERE MEALNUMBER = gvs_menu-maincourse.
      SELECT SINGLE text FROM SMEALT
        INTO gv_text_dess
        WHERE MEALNUMBER = gvs_menu-dessert.

      GS_SMENU-APP_TXT = gv_text_app.
      GS_SMENU-MAIN_TXT = gv_text_main.
      GS_SMENU-DESS_TXT = gv_text_dess.
      CALL SCREEN 110
      STARTING AT 5 5
      ENDING AT 70 9.
    ENDIF.
  ENDMETHOD.
ENDCLASS.
