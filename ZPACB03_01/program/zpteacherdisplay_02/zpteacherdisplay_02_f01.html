FORM set_data .
  SELECT * FROM ztperson_02
    WHERE ( COURSE_ID = @ZTCURRICULUM_02-COURSE_ID OR @ZTCURRICULUM_02-COURSE_ID IS INITIAL )
      AND ( GENERATION = @ZTCURRICULUM_02-GENERATION OR @ZTCURRICULUM_02-GENERATION IS INITIAL )
      AND ( NAME = @ZTPERSON_02-NAME OR @ZTPERSON_02-NAME IS INITIAL )
      AND personal_id LIKE 'T%'
    INTO CORRESPONDING FIELDS OF TABLE @gt_data.

  CLEAR: gs_data.
  LOOP AT gt_data INTO gs_data.
    CLEAR gs_history.
    SELECT POS_CODE STATUS
      FROM ZTTHISTORY_02
      UP TO 1 ROWS
      INTO CORRESPONDING FIELDS OF TABLE gt_history
      WHERE PERSONAL_ID = gs_data-personal_id
      ORDER BY start_date DESCENDING.
    READ TABLE gt_history INTO gs_history INDEX 1.
    IF gs_history-pos_code = 'M'.
      gs_data-pos_code_txt = '####'.
    ELSEIF gs_history-pos_code = 'S'.
      gs_data-pos_code_txt = '####'.
    ENDIF.

    IF gs_history-status = 'A'.
      gs_data-status_txt = '###'.
    ELSEIF gs_history-status = 'B'.
      gs_data-status_txt = '## ##'.
    ELSEIF gs_history-status = 'C'.
      gs_data-status_txt = '## ##'.
    ELSEIF gs_history-status = 'D'.
      gs_data-status_txt = '##'.
    ELSEIF gs_history-status = 'E'.
      gs_data-status_txt = '##'.
    ENDIF.

    IF gs_data-birth_date+4(4) = sy-datum+4(4).
      gs_data-birth_icon = ICON_EXTRA.
    ENDIF.

    MODIFY gt_data FROM gs_data TRANSPORTING pos_code_txt status_txt birth_icon.
    CLEAR gs_data.
  ENDLOOP.
ENDFORM.
