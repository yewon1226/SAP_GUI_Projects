REPORT yprogb03_assignment1.

PARAMETERS: pa_cour TYPE zstcourse_b03.

DATA: gs_data TYPE ztraining_cb03,
      gt_Data LIKE TABLE OF gs_data.

SELECT * FROM ztraining_cb03
  INTO TABLE gt_data WHERE course_id = pa_cour.

LOOP AT gt_data INTO gs_data.
  WRITE:/ gs_data-stu_id,
          gs_data-course_id.
  IF gs_data-class = 'C2'.
    WRITE gs_data-class COLOR COL_TOTAL.
  ELSE.
    WRITE gs_data-class.
  ENDIF.
  WRITE: gs_data-status,
         gs_data-fname,
         gs_data-lname.
ENDLOOP.
