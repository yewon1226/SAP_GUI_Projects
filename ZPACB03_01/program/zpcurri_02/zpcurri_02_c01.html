CLASS lcl_event_handler DEFINITION.
  PUBLIC SECTION.
    CLASS-METHODS on_toolbar
      FOR EVENT toolbar OF cl_gui_alv_grid
      IMPORTING e_object.

    CLASS-METHODS on_user_command
      FOR EVENT user_command OF cl_gui_alv_grid
      IMPORTING e_ucomm.
ENDCLASS.

CLASS lcl_event_handler IMPLEMENTATION.
  METHOD on_toolbar.
    DATA ls_button TYPE STB_BUTTON.
    ls_button-butn_type = 3.
    INSERT ls_button INTO TABLE e_object->mt_toolbar.

    CLEAR ls_button.
    ls_button-function = 'CREATE'.
    ls_button-butn_type = 0.
    ls_button-text = '##'.
    INSERT ls_button INTO TABLE e_object->mt_toolbar.

    CLEAR ls_button.
    ls_button-function = 'UPDATE'.
    ls_button-butn_type = 0.
    ls_button-text = '##'.
    INSERT ls_button INTO TABLE e_object->mt_toolbar.

    CLEAR ls_button.
    ls_button-function = 'DELETE'.
    ls_button-butn_type = 0.
    ls_button-text = '##'.
    INSERT ls_button INTO TABLE e_object->mt_toolbar.
  ENDMETHOD.

  METHOD on_user_command.
    CLEAR gs_data.
    READ TABLE gt_data INTO gs_data INDEX lines( gt_data ).

    CASE E_UCOMM.
      WHEN 'CREATE'.
        IF ZTCURRICULUM_02-COURSE_ID IS INITIAL OR ZTCURRICULUM_02-GENERATION IS INITIAL.
          MESSAGE i001(ZMCCL2_02).
        ELSEIF gv_btn_flag = 0.
          MESSAGE i107(ZMCCL2_02).
        ELSEIF gs_data-end_date < sy-datum.
          MESSAGE i106(ZMCCL2_02).
        ELSE.
          PERFORM create_data.
        ENDIF.
      WHEN 'UPDATE'.
        IF ZTCURRICULUM_02-COURSE_ID IS INITIAL OR ZTCURRICULUM_02-GENERATION IS INITIAL.
          MESSAGE i001(ZMCCL2_02).
        ELSEIF gv_btn_flag = 0.
          MESSAGE i107(ZMCCL2_02).
        ELSEIF gs_data-end_date < sy-datum.
          MESSAGE i106(ZMCCL2_02).
        ELSE.
          PERFORM ACTIVATE_DATA.
        ENDIF.
      WHEN 'DELETE'.
        IF ZTCURRICULUM_02-COURSE_ID IS INITIAL OR ZTCURRICULUM_02-GENERATION IS INITIAL.
          MESSAGE i001(ZMCCL2_02).
        ELSEIF gv_btn_flag = 0.
          MESSAGE i107(ZMCCL2_02).
        ELSEIF gs_data-end_date < sy-datum.
          MESSAGE i106(ZMCCL2_02).
        ELSE.
          PERFORM delete_data.
        ENDIF.
      WHEN OTHERS.
    ENDCASE.
  ENDMETHOD.
ENDCLASS.
