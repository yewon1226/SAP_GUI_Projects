  SELECT * FROM ztcurriculum_02
    INTO CORRESPONDING FIELDS OF TABLE gt_data
    WHERE course_id = ztcurriculum_02-course_id
      AND generation = ztcurriculum_02-generation.
  IF sy-subrc <> 0.
    MESSAGE i002(zmccl2_02).
  ELSE.
    CLEAR gs_data.
    LOOP AT gt_data INTO gs_data.
      gs_data-week = gv_cnt && '##'.
      MODIFY gt_data FROM gs_data TRANSPORTING week.
      gv_cnt = gv_cnt + 1.
      CLEAR gs_data.
    ENDLOOP.
  ENDIF.

  "### ##

    CLEAR gs_data4.

    APPEND gs_data4 TO gt_data4.

FORM activate_data.
  READ TABLE gt_data4 INTO gs_data4
  WITH KEY course_id = ztcurriculum_02-course_id
           generation = ztcurriculum_02-generation.

*******************************************************

  DATA : lt_index_rows TYPE lvc_t_row,
         ls_index_row  TYPE lvc_s_row.

  CALL METHOD go_alv->get_selected_rows
    IMPORTING
      et_index_rows = lt_index_rows.

  IF lt_index_rows IS INITIAL.
    MESSAGE '### #### #####' TYPE 'I'.
    EXIT.
  ENDIF.

  "### ##### ###### ## ###
  LOOP AT gt_data4 INTO gs_data4 WHERE it_cell_style IS NOT INITIAL.
    DELETE gs_data4-it_cell_style WHERE fieldname = 'START_DATE'.
    DELETE gs_data4-it_cell_style WHERE fieldname = 'END_DATE'.
    DELETE gs_data4-it_cell_style WHERE fieldname = 'CURRICULUM'.
    MODIFY gt_data4 FROM gs_data4 TRANSPORTING it_cell_style.
  ENDLOOP.

  "### Row #### #### ##
  LOOP AT lt_index_rows INTO ls_index_row.
    CLEAR : gs_data4, gs_cell_style.
    READ TABLE gt_data4 INTO gs_data4 INDEX ls_index_row-index.
    IF sy-subrc = 0.
      gs_cell_style-fieldname = 'START_DATE'.
      gs_cell_style-style = cl_gui_alv_grid=>mc_style_enabled.
      INSERT gs_cell_style INTO TABLE gs_data4-it_cell_style.

      gs_cell_style-fieldname = 'END_DATE'.
      gs_cell_style-style = cl_gui_alv_grid=>mc_style_enabled.
      INSERT gs_cell_style INTO TABLE gs_data4-it_cell_style.

      gs_cell_style-fieldname = 'CURRICULUM'.
      gs_cell_style-style = cl_gui_alv_grid=>mc_style_enabled.
      INSERT gs_cell_style INTO TABLE gs_data4-it_cell_style.

      MODIFY gt_data4 FROM gs_data4 INDEX ls_index_row-index
      TRANSPORTING it_cell_style.
    ENDIF.
  ENDLOOP.

  " Edit ###
  CALL METHOD go_alv->set_ready_for_input
    EXPORTING
      i_ready_for_input = 1. "## ## 1 #### edit# ###

FORM delete_data .
  DATA : lt_data LIKE gt_data3,
         ls_data LIKE gs_data3.
  DATA : lt_select_data TYPE lvc_t_row,
         ls_select_data TYPE lvc_s_row.


  CALL METHOD go_alv->get_selected_rows
    IMPORTING
      et_index_rows = lt_select_data.

  IF lt_select_data IS NOT INITIAL.
    LOOP AT lt_select_data INTO ls_select_data.
      READ TABLE gt_data INTO ls_data
      INDEX ls_select_data-index.

      IF sy-subrc = 0.
        APPEND ls_data TO lt_data.
        CLEAR ls_data.
      ENDIF.
    ENDLOOP.
    "## ##
    CALL FUNCTION 'POPUP_TO_CONFIRM'
      EXPORTING
        titlebar       = 'Information'
        text_question  = '## ########??'
      IMPORTING
        answer         = gv_answer
      EXCEPTIONS
        text_not_found = 1
        OTHERS         = 2.

    IF gv_answer = '1'.
      DELETE ztcurriculum_02 FROM TABLE lt_data.
      CLEAR: lt_data, ls_data.
    ELSE.
    ENDIF.
    PERFORM set_data.
    PERFORM refresh.
  ELSE.
    MESSAGE 'Please Select record' TYPE 'E'.
  ENDIF.
ENDFORM.

FORM update_data .
  " ======================================
  " ## ##
  " ======================================
  DATA: lv_index1 TYPE sy-tabix,
        lv_index2 TYPE sy-tabix,
        ls_cmp    LIKE gs_data4,        " ### WA
        ls_db     LIKE ztcurriculum_02,
        lt_data   TYPE STANDARD TABLE OF ztcurriculum_02.

  " ======================================
  " 1. ALV ## ### ##
  " ======================================
  CALL METHOD go_alv->check_changed_data.

  " ======================================
  " 2. ALV ## row ## ##
  " ======================================
  DO LINES( gt_data4 ) TIMES.
    lv_index1 = sy-index.

    " ## row ##
    READ TABLE gt_data4 INTO gs_data4 INDEX lv_index1.
    IF sy-subrc <> 0.
      CONTINUE.
    ENDIF.

    " ### ##
    IF gs_data4-start_date IS INITIAL
    OR gs_data4-end_date   IS INITIAL
    OR gs_data4-curriculum IS INITIAL.
      MESSAGE '# ####' TYPE 'E'.
    ENDIF.

    " ## ## ##
    IF gs_data4-start_date > gs_data4-end_date.
      MESSAGE '#### ##### #' TYPE 'E'.
    ENDIF.

    " ======================================
    " 2# ##: ## row# ## ##
    " ======================================
    DO LINES( gt_data4 ) TIMES.
      lv_index2 = sy-index.

      " ## ## ##
      IF lv_index1 = lv_index2.
        CONTINUE.
      ENDIF.

      " ### row ##
      READ TABLE gt_data4 INTO ls_cmp INDEX lv_index2.
      IF sy-subrc <> 0.
        CONTINUE.
      ENDIF.

      " ## course_id, generation# ##
      IF ls_cmp-course_id  = gs_data4-course_id
     AND ls_cmp-generation = gs_data4-generation.

        " ### row ##
        IF ls_cmp-start_date IS INITIAL
        OR ls_cmp-end_date   IS INITIAL.
          CONTINUE.
        ENDIF.

        " ## ## ##
        IF gs_data4-start_date <= ls_cmp-end_date
       AND gs_data4-end_date   >= ls_cmp-start_date.
          MESSAGE '### ### ####' TYPE 'E'.
        ENDIF.

      ENDIF.

    ENDDO.

  ENDDO.

  " ======================================
  " 3. DB ### ### ##
  " ======================================
  CLEAR lt_data.
  LOOP AT gt_data4 INTO gs_data4.
    CLEAR ls_db.
    MOVE-CORRESPONDING gs_data4 TO ls_db.
    ls_db-mandt = sy-mandt.
    APPEND ls_db TO lt_data.
  ENDLOOP.

  " ======================================
  " 4. DB UPDATE
  " ======================================
  MODIFY ztcurriculum_02 FROM TABLE lt_data.
  COMMIT WORK.
  CLEAR lt_data.

ENDFORM.
