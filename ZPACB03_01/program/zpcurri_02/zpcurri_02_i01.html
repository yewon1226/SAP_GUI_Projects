MODULE user_command_0100 INPUT.
  CASE ok_code.
    WHEN 'BTN'.
      gv_btn_flag = 1.
      IF ztcurriculum_02-course_id IS INITIAL OR ztcurriculum_02-generation IS INITIAL.
        MESSAGE i001(zmccl2_02).
      ELSE.
        PERFORM set_data.
        PERFORM refresh.
      ENDIF.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'SAVE'.
      PERFORM UPDATE_DATA.
      PERFORM set_data.
      PERFORM REFRESH.
    WHEN OTHERS.
  ENDCASE.
ENDMODULE.

    IF gs_data2-start_date IS INITIAL AND
       gs_data2-end_date IS INITIAL AND
       gs_data2-curriculum IS INITIAL.
      CONTINUE.

       gs_data2-end_date IS NOT INITIAL AND

      IF gs_data2-start_date BETWEEN gs_data-start_date AND gs_data-end_date
         OR

        gs_data3-generation = ztcurriculum_02-generation.
