REPORT ZBC400_03_PRACTICE1.

* ## # #### ##
PARAMETERS: pa_car TYPE S_CARR_ID.




TYPES: BEGIN OF ts_fli.
        INCLUDE TYPE SFLIGHT.
TYPES: carrname     TYPE scarr-carrname,     " SCARR (## ##)
       cityfrom     TYPE spfli-cityfrom,     " SPFLI (## ##)
       cityto       TYPE spfli-cityto,       " SPFLI (## ##)
       seatremain   TYPE i,                   " Economy ## ## (## ##)
       seatrem_b    TYPE i,                   " Business ## ## (## ##)
       seatrem_f    TYPE i,                   " First ## ## (## ##)
       END OF ts_fli.
TYPES: tt_fli TYPE TABLE OF ts_fli.

*DATA: BEGIN OF gs_fli.
*        INCLUDE TYPE SFLIGHT.
*DATA: carrname     TYPE scarr-carrname,     " SCARR (## ##)
*       cityfrom     TYPE spfli-cityfrom,     " SPFLI (## ##)
*       cityto       TYPE spfli-cityto,       " SPFLI (## ##)
*       seatremain   TYPE i,                   " Economy ## ## (## ##)
*       seatrem_b    TYPE i,                   " Business ## ## (## ##)
*       seatrem_f    TYPE i,                   " First ## ## (## ##)
*       END OF gs_fli.

*DATA: gt_fli LIKE TABLE OF gs_fli.
* # ##

DATA: gs_fli TYPE ts_fli,
      gt_fli TYPE tt_fli,
      gt_scar TYPE TABLE OF SCARR,
      gs_scar LIKE LINE OF gt_scar,
      gt_spf TYPE TABLE OF SPFLI,
      gs_spf LIKE LINE OF gt_spf.

SELECT-OPTIONS: so_con FOR gs_fli-CONNID,
                so_dat FOR gs_fli-FLDATE.

* INITIALIZATION
INITIALIZATION.
  pa_car = 'LH'.
  so_dat-sign = 'I'.
  so_dat-option = 'GE'.
  so_dat-low = '20250101'.
  APPEND so_dat.

* AT SELECTION-SCREEN
AT SELECTION-SCREEN.
  IF pa_car IS INITIAL.
    MESSAGE 'carrid #### ####!' TYPE 'E'.
  ENDIF.

* START-OF-SELECTION
START-OF-SELECTION.
  PERFORM get_data.

  IF gt_fli IS INITIAL.
    MESSAGE 'Data is not found' TYPE 'I'.
    STOP.
  ENDIF.

  PERFORM process_data CHANGING gt_fli.

  SORT gt_fli BY FLDATE ASCENDING.
  ULINE.
  cl_demo_output=>display( gt_fli ).
  ULINE.
  FREE gt_fli.

* ##
  FORM get_data.
    SELECT * FROM SFLIGHT
      INTO TABLE gt_fli
      WHERE carrid = pa_car
        AND connid IN so_con
        AND fldate IN so_dat.

    IF gt_fli IS NOT INITIAL.
      SELECT carrid carrname FROM SCARR
        INTO CORRESPONDING FIELDS OF TABLE gt_scar
        FOR ALL ENTRIES IN gt_fli
        WHERE carrid = gt_fli-carrid.
      SORT gt_scar BY carrid.

      SELECT carrid connid cityfrom cityto FROM SPFLI
        INTO CORRESPONDING FIELDS OF TABLE gt_spf
        FOR ALL ENTRIES IN gt_fli
        WHERE carrid = gt_fli-carrid
          AND connid = gt_fli-connid.
      SORT gt_spf BY carrid connid.
    ENDIF.
  ENDFORM.

* FORM process_data CHANGING gt_data LIKE gt_fli. # ##
  FORM process_data CHANGING gt_data TYPE tt_fli.
    DATA: gs_data LIKE LINE OF gt_data.

    LOOP AT gt_data INTO gs_data.
      READ TABLE gt_scar INTO gs_scar WITH KEY carrid = gs_data-carrid BINARY SEARCH.
      IF sy-subrc = 0.
        gs_data-carrname = gs_scar-carrname.
      ENDIF.
      MODIFY gt_data FROM gs_data.

      READ TABLE gt_spf INTO gs_spf
        WITH KEY carrid = gs_data-carrid
                 connid = gs_data-connid BINARY SEARCH.
      IF sy-subrc = 0.
        gs_data-cityfrom = gs_spf-cityfrom.
        gs_data-cityto = gs_spf-cityto.
      ENDIF.
      MODIFY gt_data FROM gs_data.

      gs_data-seatremain  = gs_data-seatsmax - gs_data-seatsocc.
      gs_data-seatrem_b = gs_data-seatsmax_b - gs_data-seatsocc_b.
      gs_data-seatrem_f = gs_data-seatsmax_f - gs_data-seatsocc_f.
      MODIFY gt_data FROM gs_data.

    ENDLOOP.

  ENDFORM.
