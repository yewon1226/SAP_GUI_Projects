MODULE check_sflight INPUT.

  IF OK_CODE = 'FCBTN'.
    gv_button = 1.

    SELECT SINGLE * FROM sflight
      INTO CORRESPONDING FIELDS OF gs_data
      WHERE carrid = SDYN_CONN-carrid
        AND connid = SDYN_CONN-connid
        AND fldate = SDYN_CONN-fldate.

    IF sy-subrc <> 0.
      gv_subrc = 4.
      message text-007 TYPE 'I'.
    ENDIF.
  ENDIF.

ENDMODULE.
