FORM set_data .
  IF INPUT_DATA-TRAVELID IS INITIAL.
    SELECT FROM ZDYTRAVEL_CR_B03
      FIELDS TravelId, AgencyId, Agencyname, BeginDate, EndDate, DurationDays,
             CurrencyCode, TotalPrice, TravelStatus, TravelStatusText, BookingGrade
      INTO CORRESPONDING FIELDS OF TABLE @gt_data.
  ELSE.
    SELECT FROM ZDYTRAVEL_CR_B03
      FIELDS TravelId, AgencyId, Agencyname, BeginDate, EndDate, DurationDays,
             CurrencyCode, TotalPrice, TravelStatus, TravelStatusText, BookingGrade
      WHERE travelid = @INPUT_DATA-TRAVELID
      INTO CORRESPONDING FIELDS OF TABLE @gt_data.
  ENDIF.

  SORT gt_data BY travelid.
ENDFORM.
