FORM set_data .
  DATA: gv_class_cnt TYPE i,
        gv_name TYPE CHAR10.
  DATA: gv_attend_rate TYPE P DECIMALS 2.

  DATA: lv_pid TYPE ztattedent_02-personal_id,
        lv_total_days TYPE i,
        lv_absence    TYPE i,
        lv_rl_cnt     TYPE i,
        lv_attend_rate TYPE P DECIMALS 2,
        total_attend_rate TYPE P DECIMALS 2 VALUE 0.

  SELECT * FROM ZTCOURSE_01
    INTO CORRESPONDING FIELDS OF TABLE gt_data
    WHERE COURSE_ID = ZTCURRICULUM_02-COURSE_ID
      AND GENERATION = ZTCURRICULUM_02-GENERATION.
  IF sy-subrc <> 0.
    MESSAGE i002(ZMCCL2_02).
  ENDIF.

  CLEAR gs_data.
  LOOP AT gt_data INTO gs_data.

********** ### ####
    SELECT * FROM ztthistory_02
      UP TO 1 ROWS
      INTO TABLE gt_t_history
      WHERE COURSE_ID = ZTCURRICULUM_02-COURSE_ID
        AND GENERATION = ZTCURRICULUM_02-GENERATION
        AND CLASS_NO = gs_data-class_no
        AND POS_CODE = 'M'
      ORDER BY START_DATE DESCENDING.
    READ TABLE gt_t_history INTO gs_t_history INDEX 1.

    SELECT SINGLE NAME FROM ZTPERSON_01
      INTO gv_name
      WHERE personal_id = gs_t_history-personal_id.
    gs_data-m_name = gv_name.
    MODIFY gt_data FROM gs_data TRANSPORTING m_name.
    CLEAR: gv_name, gt_t_history, gs_t_history.

    SELECT * FROM ztthistory_02
      UP TO 1 ROWS
      INTO TABLE gt_t_history
      WHERE COURSE_ID = ZTCURRICULUM_02-COURSE_ID
        AND GENERATION = ZTCURRICULUM_02-GENERATION
        AND CLASS_NO = gs_data-class_no
        AND POS_CODE = 'S'
      ORDER BY START_DATE DESCENDING.
    READ TABLE gt_t_history INTO gs_t_history INDEX 1.

    SELECT SINGLE NAME FROM ZTPERSON_01
      INTO gv_name
      WHERE personal_id = gs_t_history-personal_id.
    gs_data-s_name = gv_name.
    MODIFY gt_data FROM gs_data TRANSPORTING s_name.
    CLEAR: gv_name, gt_t_history, gs_t_history.

********** ###
    SELECT * FROM ZTSHISTORY_01
      INTO TABLE gt_s_history
      WHERE course_id  = ZTCURRICULUM_02-course_id
        AND generation = ZTCURRICULUM_02-generation
        AND class_no   = gs_data-class_no
        AND status     IN ('I','C').
    SORT gt_s_history BY personal_id.
    DELETE ADJACENT DUPLICATES FROM gt_s_history COMPARING personal_id.

    IF gt_s_history IS NOT INITIAL.
      SELECT * FROM ZTPERSON_01
        INTO CORRESPONDING FIELDS OF TABLE gt_students
        FOR ALL ENTRIES IN gt_s_history
        WHERE personal_id = gt_s_history-personal_id.
    ENDIF.

    gs_data-count_stu = lines( gt_students ).
    MODIFY gt_data FROM gs_data TRANSPORTING count_stu.
    CLEAR: gt_students, gt_s_history.

********** ###
    SELECT DISTINCT personal_id
      INTO CORRESPONDING FIELDS OF TABLE gt_sattend
      FROM ztattedent_02
      WHERE course_id  = ZTCURRICULUM_02-course_id
        AND generation = ZTCURRICULUM_02-generation
        AND class_no   = gs_data-class_no.

    CLEAR total_attend_rate.
    LOOP AT gt_sattend INTO gs_sattend.
      CLEAR: lv_total_days, lv_absence, lv_rl_cnt, lv_attend_rate.

      SELECT COUNT(*)
       INTO lv_total_days
       FROM ztattedent_02
       WHERE personal_id = gs_sattend-personal_id
         AND course_id  = ZTCURRICULUM_02-course_id
         AND generation = ZTCURRICULUM_02-generation
         AND class_no   = gs_data-class_no.

      SELECT COUNT(*)
        INTO lv_absence
        FROM ztattedent_02
        WHERE personal_id = gs_sattend-personal_id
          AND status = 'X'
          AND course_id  = ZTCURRICULUM_02-course_id
          AND generation = ZTCURRICULUM_02-generation
          AND class_no   = gs_data-class_no.

      SELECT COUNT(*)
        INTO lv_rl_cnt
        FROM ztattedent_02
        WHERE personal_id = gs_sattend-personal_id
          AND status IN ('R','L')
          AND course_id  = ZTCURRICULUM_02-course_id
          AND generation = ZTCURRICULUM_02-generation
          AND class_no   = gs_data-class_no.

      lv_absence = lv_absence + ( lv_rl_cnt / 3 ).

      " ### ##
      IF lv_total_days > 0.
        lv_attend_rate = ( lv_total_days - lv_absence ) * 100 / lv_total_days.
        total_attend_rate = total_attend_rate + lv_attend_rate.
      ENDIF.
    ENDLOOP.

    gs_data-attend_rate = total_attend_rate / lines( gt_sattend ).
    MODIFY gt_data FROM gs_data TRANSPORTING attend_rate.

********** ###
   SELECT * FROM ZTSHISTORY_01
     INTO CORRESPONDING FIELDS OF TABLE gt_s_history
     WHERE course_id  = ZTCURRICULUM_02-course_id
       AND generation = ZTCURRICULUM_02-generation
       AND class_no   = gs_data-class_no
       AND GETJOB = 'O'.
   gs_data-emp_rate = lines( gt_s_history ) * 100 / lines( gt_sattend ).
   MODIFY gt_data FROM gs_data TRANSPORTING emp_rate.

   CLEAR: gt_s_history, gt_sattend.
  ENDLOOP.

  SORT gt_data BY class_no.
ENDFORM.

FORM go_120_screen .
  SELECT DISTINCT h~personal_id, p~name
    INTO TABLE @gt_teacher_120
    FROM ZTTHISTORY_02 AS h
    INNER JOIN ZTPERSON_01 AS p
            ON h~personal_id = p~personal_id
    WHERE h~pos_code = @gv_pos_code.

  CALL SCREEN 120
    STARTING AT 10 5
    ENDING   AT 70 15.

  CLEAR OK_CODE.
ENDFORM.

FORM save_data .
  CALL METHOD go_alv_300->check_changed_data.

  IF gt_data_300 IS INITIAL.
    MESSAGE i101(zmccl2_02).
    RETURN.
  ENDIF.

  DATA: init_flag TYPE I VALUE 0.
  DATA: lv_max_class TYPE ZTCOURSE_01-class_no.
  DATA: lv_cnt TYPE i VALUE 0,
        lv_max_id   TYPE ZTSHISTORY_01-history_id,
        lv_num_part TYPE n LENGTH 9,
        lv_curr_num TYPE i,
        lv_curr_num_c TYPE n LENGTH 9,
        lv_new_id   TYPE ZTSHISTORY_01-history_id..

  DATA: lv_numc9 TYPE n LENGTH 9,
        lv_class_no TYPE NUMC2.
  DATA: lt_tea_history TYPE TABLE OF ztthistory_02,
        ls_m_tea_history TYPE ZTTHISTORY_02,
        ls_s_tea_history TYPE ZTTHISTORY_02.
  DATA: ls_course TYPE ZTCOURSE_01.
  DATA: ls_student TYPE ZTPERSON_01,
        lt_student TYPE TABLE OF ZTPERSON_01.
  DATA: lt_stu_history TYPE TABLE OF ZTSHISTORY_01,
        ls_stu_history TYPE ZTSHISTORY_01.
  DATA: ls_mteacher TYPE ZTPERSON_01,
        ls_steacher TYPE ZTPERSON_01.

  SELECT MAX( class_no )
    INTO lv_max_class
    FROM ZTCOURSE_01
    WHERE course_id  = ZTCURRICULUM_02-course_id
      AND generation = ZTCURRICULUM_02-generation.

********* ## ##, #### ### #### *************************************************************
  " ### ##
  SELECT SINGLE * FROM ZTPERSON_01
    INTO ls_mteacher
    WHERE personal_id = pa_mcode.
  ls_mteacher-course_id = ZTCURRICULUM_02-course_id.
  ls_mteacher-generation = ZTCURRICULUM_02-generation.
  ls_mteacher-class_no = lv_max_class + 1.

  " #### ##
  SELECT SINGLE * FROM ZTPERSON_01
    INTO ls_steacher
    WHERE personal_id = pa_scode.
  ls_steacher-course_id = ZTCURRICULUM_02-course_id.
  ls_steacher-generation = ZTCURRICULUM_02-generation.
  ls_steacher-class_no = lv_max_class + 1.

  " ### ## ####
  ls_m_tea_history-personal_id = pa_mcode.
  ls_m_tea_history-course_id = ZTCURRICULUM_02-course_id.
  ls_m_tea_history-generation = ZTCURRICULUM_02-generation.
  ls_m_tea_history-class_no = lv_max_class + 1. " # ####
  ls_m_tea_history-start_date = sy-datum.
  ls_m_tea_history-end_date = '99991231'.
  ls_m_tea_history-pos_code = 'M'.
  ls_m_tea_history-status = 'A'.

" #### ## ####
  ls_s_tea_history-personal_id = pa_scode.
  ls_s_tea_history-course_id = ZTCURRICULUM_02-course_id.
  ls_s_tea_history-generation = ZTCURRICULUM_02-generation.
  ls_s_tea_history-class_no = lv_max_class + 1. " # ####
  ls_s_tea_history-start_date = sy-datum.
  ls_s_tea_history-end_date = '99991231'.
  ls_s_tea_history-pos_code = 'S'.
  ls_s_tea_history-status = 'A'.

  LOOP AT gt_data_300 INTO gs_data_300.
    IF gs_data_300-name      IS INITIAL AND
      gs_data_300-gender     IS INITIAL AND
      gs_data_300-mobile     IS INITIAL AND
      gs_data_300-email      IS INITIAL AND
      gs_data_300-birth_date IS INITIAL AND
      gs_data_300-major      IS INITIAL.
      CONTINUE.
    ENDIF.

    IF gs_data_300-name      IS INITIAL OR
      gs_data_300-gender     IS INITIAL OR
      gs_data_300-mobile     IS INITIAL OR
      gs_data_300-email      IS INITIAL OR
      gs_data_300-birth_date IS INITIAL OR
      gs_data_300-major      IS INITIAL.

     MESSAGE i101(zmccl2_02).
     RETURN.
    ENDIF.
  ENDLOOP.

********* #### ### #### *************************************************************
  SELECT SINGLE * FROM ZTCOURSE_01
    INTO ls_course
    WHERE course_id  = ZTCURRICULUM_02-course_id
      AND generation = ZTCURRICULUM_02-generation.
  ls_course-class_no = lv_max_class + 1.

  LOOP AT gt_data_300 INTO gs_data_300.
    IF gs_data_300 IS NOT INITIAL.
      lv_cnt = lv_cnt + 1.
    ENDIF.
    CLEAR gs_data_300.
  ENDLOOP.
  ls_course-count_stu = lv_cnt.

  ls_course-class_location = pa_loc.

********* #######, ####(##) ### #### *************************************************************
  CLEAR gs_data_300.
  SELECT MAX( history_id ) FROM ZTSHISTORY_01
    INTO lv_max_id.
  lv_curr_num = lv_max_id+1(9).

  LOOP AT gt_data_300 INTO gs_data_300.
    IF gs_data_300 IS INITIAL.
      CONTINUE.
    ENDIF.

    MOVE-CORRESPONDING gs_data_300 TO ls_student.
    ls_student-course_id = ZTCURRICULUM_02-course_id.
    ls_student-generation = ZTCURRICULUM_02-generation.
    ls_student-class_no = lv_max_class + 1.

    lv_curr_num = lv_curr_num + 1.
    lv_curr_num_c = lv_curr_num.
    lv_new_id = |H{ lv_curr_num_c }|.
    ls_stu_history-history_id = lv_new_id.
    ls_stu_history-history_id = lv_new_id.
    ls_stu_history-course_id = ZTCURRICULUM_02-course_id.
    ls_stu_history-generation = ZTCURRICULUM_02-generation.
    ls_stu_history-class_no = lv_max_class + 1.
    ls_stu_history-start_date = sy-datum.
    ls_stu_history-end_date   = ls_s_tea_history-end_date.
    ls_stu_history-status = 'I'.
    ls_stu_history-getjob = 'X'.

    CALL FUNCTION 'NUMBER_GET_NEXT'
      EXPORTING
        nr_range_nr   = '01'
        object        = 'ZNRS02'
        quantity      = '1'
        ignore_buffer = 'X'
      IMPORTING
        number        = lv_numc9
      EXCEPTIONS
        OTHERS        = 1.

    IF sy-subrc <> 0.
      MESSAGE 'Number Range ## ##' TYPE 'E'.
    ELSE.
      ls_student-personal_id = |S{ lv_numc9 }|.
      ls_stu_history-personal_id = ls_student-personal_id.
      CLEAR lv_numc9.
    ENDIF.

    APPEND ls_student TO lt_student.
    APPEND ls_stu_history TO lt_stu_history.

    CLEAR: gs_data_300, ls_student, ls_stu_history.
  ENDLOOP.

*********** DB ### ####  ***********************************************************

  INSERT ZTPERSON_01 FROM ls_mteacher. " ##, ##
  INSERT ZTPERSON_01 FROM ls_steacher.
  INSERT ZTTHISTORY_02 FROM ls_m_tea_history.
  INSERT ZTTHISTORY_02 FROM ls_s_tea_history.

  INSERT ZTCOURSE_01 FROM ls_course. " ####

  INSERT ZTPERSON_01 FROM TABLE lt_student. " ##, ##
  INSERT ZTSHISTORY_01 FROM TABLE lt_stu_history.
ENDFORM.
