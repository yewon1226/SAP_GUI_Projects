ENDCLASS.

   METHOD on_user_command.
    DATA: lt_select_data TYPE LVC_T_ROW,
          ls_select_data TYPE LVC_S_ROW.
    DATA: lv_end_date TYPE ZTCOURSE_01-end_date.

    IF sender = go_alv.
      CASE e_ucomm.
        WHEN 'CREATE'.
          IF ZTCURRICULUM_02-COURSE_ID IS INITIAL OR ZTCURRICULUM_02-GENERATION IS INITIAL.
            MESSAGE i001(ZMCCL2_02).
          ELSEIF gv_btn_flag = 0.
            MESSAGE i107(ZMCCL2_02).
          ELSE.
            SELECT SINGLE end_date FROM ZTCOURSE_01
              INTO lv_end_date
              WHERE course_id = ZTCURRICULUM_02-COURSE_ID
                AND generation = ZTCURRICULUM_02-GENERATION.
            IF lv_end_date < sy-datum.
              MESSAGE i109(ZMCCL2_02).
            ELSE.
              CALL SCREEN 300.
              CLEAR OK_CODE.
            ENDIF.
          ENDIF.
        WHEN 'UPDATE'.
          IF ZTCURRICULUM_02-COURSE_ID IS INITIAL OR ZTCURRICULUM_02-GENERATION IS INITIAL.
            MESSAGE i001(ZMCCL2_02).
          ELSEIF gv_btn_flag = 0.
            MESSAGE i107(ZMCCL2_02).
          ELSE.
            SELECT SINGLE end_date FROM ZTCOURSE_01
              INTO lv_end_date
              WHERE course_id = ZTCURRICULUM_02-COURSE_ID
                AND generation = ZTCURRICULUM_02-GENERATION.
            IF lv_end_date < sy-datum.
              MESSAGE i109(ZMCCL2_02).
            ELSE.
              CALL METHOD go_alv->get_selected_rows
                IMPORTING
                  et_index_rows = lt_select_data.
              IF lines( lt_select_data ) = 0 OR lines( lt_select_data ) > 1.
                MESSAGE i000(ZMCCL2_02).
              ELSE.
                READ TABLE lt_select_data INTO ls_select_data INDEX 1.
                gv_update_index = ls_select_data-index.
                PERFORM set_200_data.
                CALL SCREEN 200.
                CLEAR OK_CODE.
              ENDIF.
            ENDIF.
          ENDIF.
        WHEN 'DELETE'.
          IF ZTCURRICULUM_02-COURSE_ID IS INITIAL OR ZTCURRICULUM_02-GENERATION IS INITIAL.
            MESSAGE i001(ZMCCL2_02).
          ELSEIF gv_btn_flag = 0.
            MESSAGE i107(ZMCCL2_02).
          ELSE.
            SELECT SINGLE end_date FROM ZTCOURSE_01
              INTO lv_end_date
              WHERE course_id = ZTCURRICULUM_02-COURSE_ID
                AND generation = ZTCURRICULUM_02-GENERATION.
            IF lv_end_date < sy-datum.
              MESSAGE i109(ZMCCL2_02).
            ELSE.
              CALL METHOD go_alv->get_selected_rows
                IMPORTING
                  et_index_rows = lt_select_data.
              IF lines( lt_select_data ) = 0.
                MESSAGE i000(ZMCCL2_02).
              ELSE.
                PERFORM delete_data USING lt_select_data ls_select_data.
                PERFORM set_data.
                CALL METHOD go_alv->refresh_table_display.
                MESSAGE i104(ZMCCL2_02).
              ENDIF.
            ENDIF.
          ENDIF.
        WHEN 'SEARCH'.
          IF ZTCURRICULUM_02-COURSE_ID IS INITIAL OR ZTCURRICULUM_02-GENERATION IS INITIAL.
            MESSAGE i001(ZMCCL2_02).
          ELSEIF gv_btn_flag = 0.
            MESSAGE i107(ZMCCL2_02).
          ELSE.
            CALL METHOD go_alv->get_selected_rows
              IMPORTING
                et_index_rows = lt_select_data.

            IF lines( lt_select_data ) = 0 OR lines( lt_select_data ) > 1.
              MESSAGE i000(ZMCCL2_02).
            ELSE.
              READ TABLE lt_select_data INTO ls_select_data INDEX 1.
              gv_search_index = ls_select_data-index.
              PERFORM set_screen110_data.
              CALL SCREEN 110
               STARTING AT 5  3
               ENDING   AT 90 15.
              CLEAR OK_CODE.
            ENDIF.
          ENDIF.
        WHEN OTHERS.
      ENDCASE.
    ELSE.
      CASE e_ucomm.
        WHEN 'CREATE'.
          CALL SCREEN 400.
          CLEAR OK_CODE.
        WHEN 'BTN_UPDATE'.
          CALL METHOD go_alv_200->get_selected_rows
            IMPORTING
              et_index_rows = lt_select_data.
          IF lines( lt_select_data ) = 0.
            MESSAGE i000(ZMCCL2_02).
          ELSE.
            gv_update_success = 1.
            PERFORM update_data_200 USING lt_select_data ls_select_data.
          ENDIF.
        WHEN 'UPDATE_COMPLETE'.
          IF gv_update_success = 1.
            gv_update_success = 0.
            gv_save_success = 1.
          ENDIF.
          PERFORM update_result_data.
        WHEN 'DELETE'.
          CALL METHOD go_alv_200->get_selected_rows
            IMPORTING
              et_index_rows = lt_select_data.
          IF lines( lt_select_data ) = 0.
            MESSAGE i000(ZMCCL2_02).
          ELSE.
            PERFORM delete_data_200 USING lt_select_data ls_select_data.
            MESSAGE i104(ZMCCL2_02).
          ENDIF.
        WHEN OTHERS.
      ENDCASE.
    ENDIF.
  ENDMETHOD.

  METHOD on_double_click.
    IF sender <> go_alv_120.
      RETURN.
    ENDIF.

    CLEAR gs_teacher_120.
    READ TABLE gt_teacher_120 INTO gs_teacher_120 INDEX e_row-index.
    IF sy-subrc <> 0.
      RETURN.
    ENDIF.

    IF gv_pos_code = 'M'.
      IF screen300_go120_flag = 1.
        screen300_go120_flag = 0.
        PA_MNAME = gs_teacher_120-name.
        pa_mcode = gs_teacher_120-personal_id.
      ELSE.
        PA_DATA_200-PA_MNAME2 = gs_teacher_120-name.
        pa_mcode2 = gs_teacher_120-personal_id.
      ENDIF.
    ELSEIF gv_pos_code = 'S'.
      IF screen300_go120_flag = 1.
        screen300_go120_flag = 0.
        PA_SNAME = gs_teacher_120-name.
        pa_scode = gs_teacher_120-personal_id.
      ELSE.
        PA_DATA_200-PA_SNAME2 = gs_teacher_120-name.
        pa_scode2 = gs_teacher_120-personal_id.
      ENDIF.
    ENDIF.

    LEAVE TO SCREEN 0.
  ENDMETHOD.
ENDCLASS.
