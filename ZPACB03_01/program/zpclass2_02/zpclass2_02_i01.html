MODULE user_command_0200 INPUT.
  CASE OK_CODE.
    WHEN 'SAVE'.
      IF gv_save_success = 1.
        gv_save_success = 0.
        gv_update_success = 0.
        PERFORM save_data_200.
        PERFORM set_data.
        CALL METHOD go_alv->refresh_table_display.
        MESSAGE i102(zmccl2_02).
*        LEAVE TO SCREEN 0.
      ELSEIF gv_save_success = 0 AND gv_update_success = 1.
        gv_update_success = 0.
        MESSAGE i112(zmccl2_02).
      ELSE.
        gv_save_success = 0.
        gv_update_success = 0.
        PERFORM save_data_200.
        PERFORM set_data.
        CALL METHOD go_alv->refresh_table_display.
        MESSAGE i113(zmccl2_02).
      ENDIF.
    WHEN 'BTN01'.
      gv_pos_code = 'M'.
      PERFORM go_120_screen.
    WHEN 'BTN02'.
      gv_pos_code = 'S'.
      PERFORM go_120_screen.
    WHEN 'BACK'.
      gv_from_400 = 0.
      CLEAR gv_update_index.
      LEAVE TO SCREEN 0.
    WHEN OTHERS.
  ENDCASE.
ENDMODULE.

MODULE user_command_0300 INPUT.
  IF go_alv_300 IS NOT INITIAL.
    CALL METHOD go_alv_300->check_changed_data.
  ENDIF.

  CASE OK_CODE.
    WHEN 'SAVE'.
      IF pa_mname IS INITIAL OR pa_sname IS INITIAL OR
         pa_loc IS INITIAL.
        MESSAGE i101(zmccl2_02).
      ELSE.
        PERFORM save_data.
        PERFORM set_data.
        CALL METHOD go_alv->refresh_table_display.
        MESSAGE i100(zmccl2_02).
      ENDIF.
    WHEN 'BTN01'.
      gv_pos_code = 'M'.
      screen300_go120_flag = 1.
      PERFORM go_120_screen.
    WHEN 'BTN02'.
      gv_pos_code = 'S'.
      screen300_go120_flag = 1.
      PERFORM go_120_screen.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN OTHERS.
  ENDCASE.
ENDMODULE.
