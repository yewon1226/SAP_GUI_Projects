MODULE user_command_0100 INPUT.
  CASE OK_CODE.
    WHEN 'CHK'.
      IF 'CHK02' = 'X'.
        gv_chk = 1.
      ELSE.
        gv_chk = 0.
      ENDIF.
    WHEN 'AIR'.
      PERFORM select_flight.
    WHEN 'RES'.
      PERFORM select_booking.
    WHEN 'CUS'.
      PERFORM select_customer.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'EXIT' OR 'CANCEL'.
      LEAVE PROGRAM.
    WHEN 'FCTAB01' OR 'FCTAB02' OR 'FCTAB03' OR 'FCTAB04'.
      TAB_STRIP-activetab = OK_CODE.
    WHEN OTHERS.
  ENDCASE.

  SELECT SINGLE A~CARRID B~CONNID A~CARRNAME A~CURRCODE B~COUNTRYFR B~CITYFROM B~AIRPFROM B~COUNTRYTO B~CITYTO B~AIRPTO
      FROM scarr AS A INNER JOIN spfli AS B
      on A~carrid = B~carrid
      INTO CORRESPONDING FIELDS OF gs_scarrspfli
      WHERE A~carrid = ZSASSIGNMENT1_B03-CARRID
        AND B~connid = ZSASSIGNMENT1_B03-CONNID.
  SELECT SINGLE CARRID CONNID FLDATE PRICE CURRENCY PLANETYPE SEATSMAX SEATSOCC
      FROM sflight
      INTO CORRESPONDING FIELDS OF gs_sflight
      WHERE carrid = ZSASSIGNMENT1_B03-CARRID
        AND connid = ZSASSIGNMENT1_B03-CONNID
        AND fldate = ZSASSIGNMENT1_B03-FLDATE.
    gs_sflight-SEATSREST = gs_sflight-SEATSMAX - gs_sflight-SEATSOCC.
  IF gv_chk = 1 AND ZSASSIGNMENT1_B03-BOOKID IS NOT INITIAL.
    SELECT SINGLE CARRID CONNID FLDATE BOOKID CUSTOMID CUSTTYPE ORDER_DATE CANCELLED
      FROM SBOOK
      INTO CORRESPONDING FIELDS OF gs_book
      WHERE CARRID = ZSASSIGNMENT1_B03-CARRID AND
            CONNID = ZSASSIGNMENT1_B03-CONNID AND
            BOOKID = ZSASSIGNMENT1_B03-BOOKID AND
*            FLDATE > SY-DATUM
*            ORDER BY FLDATE ASC
*            UP TO 1 ROW.
            CANCELLED = 'X'.
    IF sy-subrc <> 0.
      MESSAGE '### ######.' TYPE 'I'.
    ENDIF.
  ELSEIF gv_chk = 0 AND ZSASSIGNMENT1_B03-BOOKID IS NOT INITIAL.
    SELECT SINGLE CARRID CONNID FLDATE BOOKID CUSTOMID CUSTTYPE ORDER_DATE CANCELLED
      FROM SBOOK
      INTO CORRESPONDING FIELDS OF gs_book
      WHERE CARRID = ZSASSIGNMENT1_B03-CARRID AND
            CONNID = ZSASSIGNMENT1_B03-CONNID AND
            BOOKID = ZSASSIGNMENT1_B03-BOOKID AND
            CANCELLED = ''.
    IF sy-subrc <> 0.
      MESSAGE '### ######.' TYPE 'I'.
    ENDIF.
  ENDIF.
  SELECT SINGLE ID NAME FORM STREET POSTBOX POSTCODE CITY COUNTRY REGION TELEPHONE
      FROM SCUSTOM
      INTO CORRESPONDING FIELDS OF gs_custom
      WHERE ID = ZSASSIGNMENT1_B03-CUSTOMID.
ENDMODULE.
