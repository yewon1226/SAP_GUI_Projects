FORM select_flight.
  SELECT *
  FROM scarr AS a
  LEFT JOIN spfli AS b
   ON a~carrid = b~carrid
  INTO CORRESPONDING FIELDS OF TABLE @gt_scarrspfli
    WHERE A~CARRID = @ZSASSIGNMENT1_B03-CARRID
      AND B~CONNID = @ZSASSIGNMENT1_B03-CONNID.
  IF sy-subrc = 0.
    cl_demo_output=>display( gt_scarrspfli ).
  ELSE.
    MESSAGE 'ERROR' TYPE 'I'.
  ENDIF.
ENDFORM.

FORM select_booking.
  SELECT *
     INTO CORRESPONDING FIELDS OF TABLE gt_book
     FROM sbook
    WHERE CARRID = ZSASSIGNMENT1_B03-CARRID
      AND CONNID = ZSASSIGNMENT1_B03-CONNID
      AND FLDATE = ZSASSIGNMENT1_B03-FLDATE.
  IF sy-subrc = 0.
    cl_demo_output=>display( gt_book ).
  ELSE.
    MESSAGE '### ######.' TYPE 'I'.
  ENDIF.
ENDFORM.
