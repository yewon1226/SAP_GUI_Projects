CLASS lcl_flights DEFINITION.
  PUBLIC SECTION.
    TYPES tt_flights TYPE STANDARD TABLE OF sflight WITH NON-UNIQUE KEY carrid connid fldate.
    CLASS-METHODS get_flights IMPORTING iv_carrid      TYPE sflight-carrid
                                        iv_connid      TYPE sflight-connid
                              RETURNING VALUE(rt_list) TYPE d75aw_t_flights.
  PRIVATE SECTION.
    CLASS-METHODS read_flights IMPORTING iv_carrid         TYPE sflight-carrid
                                         iv_connid         TYPE sflight-connid
                               RETURNING VALUE(rt_flights) TYPE tt_flights.

ENDCLASS.


CLASS lcl_flights IMPLEMENTATION.
  METHOD get_flights.
    DATA ls_result  TYPE d75aw_s_flight.

    DATA(lt_flights) = read_flights( iv_carrid = iv_carrid iv_connid = iv_connid ).
    LOOP AT lt_flights ASSIGNING FIELD-SYMBOL(<line>).
      MOVE-CORRESPONDING <line> TO ls_result.
      TRY.
          ls_result-percentage = ls_result-seatsocc / ls_result-seatsmax * 100.
          APPEND ls_result TO rt_list.
        CATCH cx_sy_zerodivide INTO DATA(lx_error).
      ENDTRY.
    ENDLOOP.
  ENDMETHOD.

  METHOD read_flights.
    SELECT * FROM sflight INTO TABLE rt_flights WHERE carrid = iv_carrid AND connid = iv_connid.
  ENDMETHOD.

ENDCLASS.
