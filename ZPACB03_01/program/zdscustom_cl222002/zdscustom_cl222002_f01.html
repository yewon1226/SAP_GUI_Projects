FORM refresh_table  USING    p_go_alv TYPE REF TO cl_gui_alv_grid
                             p_gs_stable TYPE LVC_S_STBL.
  CALL METHOD p_go_alv->refresh_table_display
    EXPORTING
      is_stable      = p_gs_stable
      i_soft_refresh = ''
    EXCEPTIONS
      finished       = 1
      others         = 2
          .
  IF sy-subrc <> 0.
    MESSAGE 'refresh # ##' TYPE 'E'.
  ENDIF.
ENDFORM.

FORM delete_tb .
  DATA: lt_select_data TYPE LVC_T_ROW,
        ls_select_data TYPE LVC_S_ROW.
  DATA: gs_data TYPE ZTSCUSTOM_CL22.

  CALL METHOD go_alv->get_selected_rows
    IMPORTING
      et_index_rows = lt_select_data.

  LOOP AT lt_select_data INTO ls_select_data.
    READ TABLE gt_cust INTO gs_data INDEX ls_select_data-index.
    IF sy-subrc = 0.
      DELETE ZTSCUSTOM_CL22 FROM gs_data.
      DELETE gt_cust INDEX ls_select_data-index.
    ENDIF.
  ENDLOOP.
  CALL METHOD go_alv->refresh_table_display.
ENDFORM.
