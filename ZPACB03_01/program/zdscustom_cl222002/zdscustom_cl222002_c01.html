CLASS lcl_event_handler DEFINITION.
  PUBLIC SECTION.
  CLASS-METHODS: ON_DOUBLE_CLICK FOR EVENT DOUBLE_CLICK
    OF CL_GUI_ALV_GRID IMPORTING ES_ROW_NO,
                 ON_TOOLBAR FOR EVENT TOOLBAR
    OF CL_GUI_ALV_GRID IMPORTING E_OBJECT,
                 on_user_command FOR EVENT user_command
    OF cl_gui_alv_grid IMPORTING e_ucomm..
ENDCLASS.

CLASS lcl_event_handler IMPLEMENTATION.
  METHOD ON_DOUBLE_CLICK.
      gs_row = ES_ROW_NO.
      PERFORM set_value USING gs_row
                        CHANGING gt_cust gs_cust
                                 gt_custom gs_custom.
  ENDMETHOD.

  METHOD ON_TOOLBAR.
  DATA: ls_button TYPE STB_BUTTON.
  ls_button-butn_type = 3.
  INSERT ls_button INTO TABLE e_object->mt_toolbar.

  CLEAR ls_button.
  ls_button-function = 'CREATE_TB'.
  ls_button-icon = ICON_CREATE.
  ls_button-butn_type = 0.
  ls_button-quickinfo = 'Create'.
  INSERT ls_button INTO TABLE e_object->mt_toolbar.

  CLEAR ls_button.
  ls_button-function = 'DELETE_TB'.
  ls_button-icon = ICON_DELETE.
  ls_button-butn_type = 0.
  ls_button-quickinfo = 'Delete'.
  INSERT ls_button INTO TABLE e_object->mt_toolbar.
 ENDMETHOD.

 METHOD on_user_command.
    CASE e_ucomm.
      WHEN 'CREATE_TB'.
        PERFORM create_tb.
      WHEN 'DELETE_TB'.
        PERFORM delete_tb.
      WHEN OTHERS.
    ENDCASE.
  ENDMETHOD.
ENDCLASS.
