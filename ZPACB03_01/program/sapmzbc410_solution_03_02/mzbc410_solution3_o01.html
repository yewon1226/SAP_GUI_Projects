MODULE status_0100 OUTPUT.

  SET PF-STATUS 'S100'.

  IF gv_title = 1.
    SET TITLEBAR 'T100' WITH 'View'.
  ELSEIF gv_title = 2.
    SET TITLEBAR 'T100' WITH 'Maintain flights'.
    LOOP AT SCREEN.
      IF screen-name = 'SDYN_CONN-PLANETYPE'.
        screen-input = 1.
      ENDIF.
      MODIFY SCREEN.
    ENDLOOP.
  ELSEIF gv_title = 3.
    SET TITLEBAR 'T100' WITH 'Maintain bookings'.
  ENDIF.

  IF gv_button = 1.
    IF gv_subrc <> 0.
      CLEAR SDYN_CONN.
      gv_subrc = 0.
    ELSE.
      MOVE-CORRESPONDING gs_data TO SDYN_CONN.
    ENDIF.
  ELSE.
    gv_button = 0.
  ENDIF.

ENDMODULE.
