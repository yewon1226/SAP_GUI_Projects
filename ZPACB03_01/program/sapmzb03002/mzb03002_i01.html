MODULE user_command_0100 INPUT.
  CASE OK_CODE.
    WHEN 'CHBTN'.
      IF gv_mod = 1.
        gv_mod = 0.
      ELSE.
        gv_mod = 1.
      ENDIF.
    WHEN 'BACK' OR 'EXIT' OR 'CANCEL'.
      LEAVE TO SCREEN 0.
    WHEN OTHERS.
  ENDCASE.
ENDMODULE.
