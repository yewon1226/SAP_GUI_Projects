REPORT ZBC400_03_12_05.

DATA: gs_dept TYPE ZDEPT_TR_B03,
      gt_dept LIKE TABLE OF gs_dept,
      gs_emp TYPE ZEMPLOYEE_B03,
      gt_emp LIKE TABLE OF ZEMPLOYEE_B03,
      flag TYPE I.

PARAMETERS: gv_emp TYPE ZDEPT_TR_B03-EMP_ID,
            gv_dept TYPE ZDEPT_TR_B03-DEPT_ID,
            gv_act TYPE ZDEPT_TR_B03-E_ACTION_DATE,
            gv_memo TYPE ZDEPT_TR_B03-MEMO.

SELECT * FROM ZDEPT_TR_B03
  INTO TABLE gt_dept.
SELECT * FROM ZEMPLOYEE_B03
  INTO TABLE gt_emp.

gs_dept-EMP_ID = gv_emp.
gs_dept-DEPT_ID = gv_dept.
gs_dept-E_ACTION_DATE = gv_act.
gs_dept-MEMO = gv_memo.

flag = 1.

INSERT ZDEPT_TR_B03 FROM gs_dept.

CLEAR gs_dept.
LOOP AT gt_dept INTO gs_dept.
  IF gv_act < gs_dept-E_ACTION_DATE AND gs_dept-EMP_ID = gv_emp.
    flag = 0.
    EXIT.
  ENDIF.
ENDLOOP.

IF flag = 1.
  LOOP AT gt_emp INTO gs_emp.
    IF gs_emp-EMP_ID = gv_emp.
      gs_emp-DEPT_ID = gv_dept.
      MODIFY ZEMPLOYEE_B03 FROM gs_emp.
    ENDIF.
  ENDLOOP.
ENDIF.

SELECT * FROM ZDEPT_TR_B03
  INTO TABLE gt_dept.
SELECT * FROM ZEMPLOYEE_B03
  INTO TABLE gt_emp.

cl_demo_output=>display( gt_dept ).
cl_demo_output=>display( gt_emp ).
