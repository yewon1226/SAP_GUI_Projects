MODULE user_command_0100 INPUT.
  CASE OK_CODE.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'EXIT' OR 'CANCEL'.
      LEAVE PROGRAM.
    WHEN 'MOVE'.
      SET SCREEN 200.
      LEAVE SCREEN.
      " LEAVE TO SCREEN 200. # ##
    WHEN 'REMARK'.
*      MESSAGE 'Remark ## ##' TYPE 'I'.
      CALL SCREEN 110
      STARTING AT 5 5
      ENDING AT 50 10.
    WHEN 'BTN01'.
      IF gv_button = 1.
        gv_button = 0.
      ELSE.
        gv_button = 1.
      ENDIF.
    WHEN 'ICON'.
      IF gv_button = 1.
        gv_button = 0.
      ELSE.
        gv_button = 1.
      ENDIF.
    WHEN 'RADIO1'.
      CASE 'X'.
        WHEN RAD01.
          MESSAGE 'Radio btn 1# ##' TYPE 'I'.
        WHEN RAD02.
          MESSAGE 'Radio btn 2# ##' TYPE 'I'.
        WHEN RAD03.
          MESSAGE 'Radio btn 3# ##' TYPE 'I'.
        WHEN OTHERS.
      ENDCASE.
    WHEN 'CHECK1'.
      MESSAGE 'Check box 1# ##' TYPE 'I'.
    WHEN OTHERS.
  ENDCASE.

ENDMODULE.
