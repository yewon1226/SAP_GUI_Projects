FORM set_icon USING icon_name
              CHANGING icon_result.
  CALL FUNCTION 'ICON_CREATE'
    EXPORTING
      name = icon_name
    IMPORTING
      RESULT                      = icon_result
    EXCEPTIONS
      ICON_NOT_FOUND              = 1
      OUTPUTFIELD_TOO_SHORT       = 2
      OTHERS                      = 3
    .
  IF sy-subrc <> 0.
    MESSAGE 'Error' TYPE 'E'.
  ENDIF.
ENDFORM.
