MODULE user_command_0100 INPUT.
  CASE ok_code.
    WHEN 'DETAIL'.
      DATA: lt_select_data TYPE LVC_T_ROW,
            ls_select_data TYPE LVC_S_ROW.
      DATA: gs_data2 TYPE ZDYTRAVEL_CR_B03.
      CALL METHOD go_alv->get_selected_rows
          IMPORTING et_index_rows = lt_select_data.
      IF LINES( lt_select_data ) NE 1.
          MESSAGE '1## ######.' TYPE 'I'.
      ELSE.
        CLEAR gs_data.
        READ TABLE gt_data INTO gs_data INDEX lt_select_data[ 1 ]-index.
        SELECT SINGLE * FROM ZDYTRAVEL_CR_B03
          WHERE travelid = @gs_data-travelid
          INTO CORRESPONDING FIELDS OF @gs_data2.
        MOVE-CORRESPONDING gs_data2 TO input_data2.
        PERFORM set_data_200.
        CALL SCREEN 200.
      ENDIF.
    WHEN 'SEARCH'.
      PERFORM set_data.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN OTHERS.
  ENDCASE.
ENDMODULE.
