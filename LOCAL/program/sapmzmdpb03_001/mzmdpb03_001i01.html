MODULE user_command_0100 INPUT.
  CASE INPUT01.
    WHEN 'X'.
      SELECT SINGLE CARRID CONNID CITYFROM CITYTO
        INTO CORRESPONDING FIELDS OF SDYN_CONN
        FROM SPFLI
        WHERE CARRID = SDYN_CONN-CARRID
          AND CONNID = SDYN_CONN-CONNID.
    WHEN 'E'.
      LEAVE TO SCREEN 0.
    WHEN OTHERS.
  ENDCASE.
ENDMODULE.
