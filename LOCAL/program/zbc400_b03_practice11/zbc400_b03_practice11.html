CLASS lcl_person DEFINITION.
  PUBLIC SECTION.
    CLASS-METHODS: display_info IMPORTING iv_name TYPE CHAR20
                                          iv_age TYPE I
                                RETURNING VALUE(rv_result) TYPE CHAR40.
  PRIVATE SECTION.
    CLASS-DATA: name TYPE CHAR20 VALUE '###',
                age TYPE I VALUE 20.
ENDCLASS.

CLASS lcl_person IMPLEMENTATION.
  METHOD display_info.
    IF iv_name IS NOT INITIAL.
      name = iv_name.
      age = iv_age.
    ENDIF.
    rv_result = name && `# ### ` && age && '# ###.'.
    WRITE:/ rv_result.
  ENDMETHOD.
ENDCLASS.
