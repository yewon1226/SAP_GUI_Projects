  DATA: lv_year TYPE CHAR10,
        lv_month TYPE CHAR10,   " ##, #, # # ### ### ## ##
        lv_date TYPE CHAR10.

  CONCATENATE pv_date+0(4) '#' INTO lv_year.    " ##, #, # # ### ## ### ##
